      "def pivot_table(df):\n",
      "    '''\n",
      "    Takes a DataFrame and returns a DataFrame.\n",
      "    \n",
      "    Parameters\n",
      "    ----------\n",
      "    df: A Pandas DataFrame, e.g. pd.read_hdf('/data/database/sflights.h5', 'table')\n",
      "    \n",
      "    Returns\n",
      "    -------\n",
      "    A Pandas DataFrame. It's a pivot table of flight cancellations\n",
      "    as a function of month and Day. See assignment 12.2 instructions.\n",
      "    '''\n",
      "    \n",
      "    #Include rows who departed from ORD and had a departure delay\n",
      "    #greater than or equal to 15min\n",
      "    df = df[(df[\"depart\"] == \"ORD\") & (df[\"dDelay\"] >= 15)]\n",
      "    \n",
      "    df = df.groupby(['month', 'Day'])\n",
      "    df = df.count()\n",
      "    \n",
      "    #Select the dDelay column and reset indices\n",
      "    df = pd.DataFrame(df.dDelay)\n",
      "    df.reset_index(inplace = True)\n",
      "   \n",
      "    result = df.pivot(\"month\", \"Day\", \"dDelay\")\n",
      "    \n",
      "    return result"
